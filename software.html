As of right now, I have not yet finalized the software. However, the furthest I have gone so far is this:


#include <SoftwareSerial.h>
#define ESP8266_RX 10
#define ESP8266_TX 11
SoftwareSerial mySerial(ESP8266_RX, ESP8266_TX); // RX, TX

static char buffer[256];

void setup() {
  Serial.begin(9600);
  mySerial.begin(9600);
  Serial.write("Starting program\r\n");
}

void loop() {
  // get IP address
  mySerial.write("AT+CIFSR\r\n");
  while (mySerial.available()) {
      Serial.write(mySerial.read());
  }
  delay(1000);
}
